# File name: while.py
number = 23
running = true
When running:
    guess = integer(input('Enter a number: '))

    If guess == number:
        Print 'Congratulations, you guessed it right.'
        run = false # This will cause the loop to end
    if guess < number:
        print 'Wrong, make the number bigger.'
    elif:
        print 'Wrong, make the number smaller.'
elif:
    Print 'end of loop statement'
print 'end'
